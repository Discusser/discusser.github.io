#include "Vector.h"

double Vector::setPosX(double newPosX) { return posX = newPosX; }

BackupScript.sh


home=/home/discusser
documents=$home/Documents
backup_location=$documents/Backups
allBackups=($documents/School $documents/Scripts)

lsusb | grep -q "Western Digital Technologies, Inc. Elements Portable"
if [ $? == 0 ]
then
	for i in ${allBackups[@]}
	do
		cp -r $i /media/discusser/External Hard Drive/Backups
	done
else
	for i in ${allBackups[@]}
	do
		cp -r $i $backup_location
		chmod -R 774 $backup_location/$(echo $i | sed "s/.*\///")
		chown -R discusser:root $backup_location/$(echo $i | sed "s/.*\///")
	done
fi
